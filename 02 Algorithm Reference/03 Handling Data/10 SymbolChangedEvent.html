$[SymbolChangedEvent,T:QuantConnect.Data.Market.SymbolChangedEvent] provides notice of new ticker names for stocks, or mergers of two tickers into one. It provides the OldSymbol and NewSymbol tickers. 

def Initialize(self):
    self.SetStartDate(2014, 4, 1)
    self.SetEndDate(2014, 4, 3)
    self.SetCash(100000);
    equity = self.AddEquity("GOOG", Resolution.Daily)
    
def OnData(self, data):
    self.MarketOrder('GOOG', 10)
    
    ## Print out old and new symbol if 'GOOG' symbol has changed
    for kvp in data.SymbolChangedEvents:
        symbol = kvp.Key
        value = kvp.Value.OldSymbol
        newSymbol = kvp.Value.NewSymbol
        self.Log(str('Old symbol: ' + str(value) + " | New symbol: " + str(newSymbol)))
