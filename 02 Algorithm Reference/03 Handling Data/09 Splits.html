def OnData(self, data):
    if not self.Portfolio.Invested:
        self.Buy("MSFT", 100)
    
    ## If MSFT had a split, print out information about it
    if data.Splits.ContainsKey("MSFT"):
        ## Log split information
        spySplit = data.Splits['MSFT']
        if spySplit.Type == 0:
            self.Log('MSFT stock will split next trading day')
        if spySplit.Type == 1:
            self.Log('Split type: ' + str(spySplit.Type) + '; Split factor: ' + str(spySplit.SplitFactor) +
                    '; Reference price: ' + str(spySplit.ReferencePrice))
