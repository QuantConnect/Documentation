Data can be accessed via Slice events. Slice delivers unique events to your algorithm as they happen. We recommend saving the symbol object when you add the data for easy access to the slice later. Data is available in daily resolution. You can see an example of the slice accessor in the code below.

def OnData(self, data):
    points = data.Get(KavoutCompositeFactorBundle)
    for kvp in points:
        Log(f'Symbol: {kvp.Key} - Growth: {kvp.Value.Growth} - Low Volatility: {kvp.Value.LowVolatility}'
            f'Momentum: {kvp.Value.Momentum} - Quality: {kvp.Value.Quality} - Value Factor: {kvp.Value.ValueFactor}')

namespace QuantConnect
{    
    public override void OnData(Slice data)
    {
        var points = data.Get<KavoutCompositeFactorBundle>();
        foreach (var point in points.Values)
        {
            Log($@"Symbol: {kvp.Key} - Growth: {kvp.Value.Growth} - Low Volatility: {kvp.Value.LowVolatility}
                Momentum: {kvp.Value.Momentum} - Quality: {kvp.Value.Quality} - Value Factor: {kvp.Value.ValueFactor}");
        }
    }
}
