To add the Cross Asset Model dataset by ExtractAlpha to your algorithm, use the AddData() method to request the data. As with all datasets, you should save a reference to your symbol for easy use later in your algorithm. For detailed documentation on using custom data, see Importing Custom Data.

class ExtractAlphaCrossAssetModelDataAlgorithm(QCAlgorithm):
    def Initialize(self):
        self.SetStartDate(2019, 1, 1)
        self.SetEndDate(2020, 6, 1)
        self.SetCash(100000)

        aapl = self.AddEquity("AAPL", Resolution.Daily).Symbol
        extract_alpha_cross_asset_model_symbol = self.AddData(ExtractAlphaCrossAssetModel, aapl).Symbol
namespace QuantConnect
{
    public class ExtractAlphaCrossAssetModelDataAlgorithm : QCAlgorithm
    {
        public override void Initialize()
        {
            SetStartDate(2019, 1, 1);
            SetEndDate(2020, 6, 1);
            SetCash(100000);

            var aapl = AddEquity("AAPL", Resolution.Daily).Symbol;
            var extractAlphaCrossAssetModelSymbol = AddData<ExtractAlphaCrossAssetModel>(aapl).Symbol;
        }
    }
