Data can be accessed via Slice events. Slice delivers unique events to your algorithm as they happen. We recommend saving the symbol object when you add the data for easy access to slice later. Data is available in daily resolution. You can see an example of the slice accessor in the code below.

class USCashIndexAlgorithm(QCAlgorithm):
    def OnData(self, data):
        # The marketData object can be a TradeBar or a List of Tick
        marketData = data[self.vix]

        # If it's TradeBar:
        self.Log(f'Symbol: {self.vix} - Data: {marketData}')

        # If it's Tick data:
        for tick in marketData:
            self.Log(f"Time: {tick.EndTime}; Symbol: {self.symbol}; Tick: {tick}")

namespace QuantConnect
{
    public class USCashIndexAlgorithm : QCAlgorithm
    {
        public override void OnData(Slice data)
        {
            // The marketData object can be a TradeBar or a List of Tick
            var marketData = data[_vix];

            // If it's TradeBar:
            Log($"Symbol: {_vix} - Data: {marketData}");

            // If it's Tick data:
            foreach (var tick in marketData)
            {
                 Log($"Time: {tick.EndTime}; Symbol: {_vix} Tick: {tick}");
            }
        }
    }
}
