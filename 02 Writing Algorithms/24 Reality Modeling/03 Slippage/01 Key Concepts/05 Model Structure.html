Slippage models should implement the ISlippageModel interface. Extensions of the ISlippageModel interface must implement the GetSlippageApproximation method which receives a Security and Order object and returns a slippage quantity.

public class MySlippageModel : ISlippageModel {

    public decimal GetSlippageApproximation(Security asset, Order order) {
        return asset.Price*0.0001m*(decimal) Math.Log10(2*(double) order.AbsoluteQuantity);
    }
}
