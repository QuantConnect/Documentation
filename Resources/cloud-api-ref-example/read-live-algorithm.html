The following example demonstates reading a live algorithm's statistics, portfolio state, logs, chart, orders, and insights through the cloud API.

from base64 import b64encode
from hashlib import sha256
from time import time
from requests import get, post

USER_ID = 0
ORG_ID = ""
API_TOKEN = '...'
BASE_URL = 'https://www.quantconnect.com/api/v2/'
project_id = 12345678
deploy_id = "..."

def get_headers():
    # Get timestamp
    timestamp = f'{int(time())}'
    time_stamped_token = f'{API_TOKEN}:{timestamp}'.encode('utf-8')

    # Get hased API token
    hashed_token = sha256(time_stamped_token).hexdigest()
    authentication = f'{USER_ID}:{hashed_token}'.encode('utf-8')
    authentication = b64encode(authentication).decode('ascii')

    # Create headers dictionary.
    return {
        'Authorization': f'Basic {authentication}',
        'Timestamp': timestamp
    }

# Read Live Algorithm Statistics
headers = get_headers()
response = post(f'{BASE_URL}/live/read', headers = headers, data = { "projectId": project_id, "deployId": deploy_id })
response.json()

# Read Live Algorithm Charts
chart_name = "Strategy Equity"
headers = get_headers()
response = post(f'{BASE_URL}/live/chart/read', headers = headers
                data = {
                    "projectId": project_id,
                    "name": chart_name,
                    "count": 100
                    "start": 1717801200,
                    "end": 1743462000
                })
response.json()

# Read Live Algorithm Portfolio State
headers = get_headers()
response = post(f'{BASE_URL}/live/portfolio/read', headers = headers, data = { "projectId": project_id })
response.json()

# Read Live Algorithm Orders
headers = get_headers()
response = post(f'{BASE_URL}/live/orders/read', headers = headers,
                data = {
                    "projectId": project_id,
                    "start": 0,
                    "end": 100
                })
response.json()

# Read Live Algorithm Insights
headers = get_headers()
response = post(f'{BASE_URL}/live/insights/read', headers = headers,
                data = {
                    "projectId": project_id,
                    "start": 0,
                    "end": 100
                })
response.json()

# Read Live Algorithm Logs
headers = get_headers()
response = post(f'{BASE_URL}/live/logs/read', headers = headers,
                data = {
                    "projectId": project_id,
                    "algorithmId": deploy_id,
                    "format": "json",
                    "startLine": 0,
                    "endLine": 100
                })
response.json()
