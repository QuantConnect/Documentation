The following example demonstates creating, reading, updating, and deleting a file through the cloud API.

from base64 import b64encode
from hashlib import sha256
from time import time
from requests import get, post

USER_ID = 0
ORG_ID = ""
API_TOKEN = '...'
BASE_URL = 'https://www.quantconnect.com/api/v2/'
project_id = 12345678

def get_headers():
    # Get timestamp
    timestamp = f'{int(time())}'
    time_stamped_token = f'{API_TOKEN}:{timestamp}'.encode('utf-8')

    # Get hased API token
    hashed_token = sha256(time_stamped_token).hexdigest()
    authentication = f'{USER_ID}:{hashed_token}'.encode('utf-8')
    authentication = b64encode(authentication).decode('ascii')

    # Create headers dictionary.
    return {
        'Authorization': f'Basic {authentication}',
        'Timestamp': timestamp
    }

# Create File
headers = get_headers()
response = post(f'{BASE_URL}/files/create', headers = headers,
                data = {
                    "projectId": project_id,
                    "name": "utils.py",
                    "content": '''
# region imports
from AlgorithmImports import *
# endregion

class Project(QCAlgorithm):

    def Initialize(self):
        self.AddEquity("SPY", Resolution.Minute)
''' })
response.json()

# Read File
headers = get_headers()
response = post(f'{BASE_URL}/files/read', headers = headers, data = { "projectId": project_id, "includeLibraries": True })
result = response.json()
fileName = result['files'][0]['name']
content = result['files'][0]['content']

# Update File Contents
headers = get_headers()
content = content.replace("SPY", "TSLA")
response = post(f'{BASE_URL}/files/update', headers = headers,
                data = {
                    "projectId": project_id,
                    "name": "utils.py",
                    "content": content
                })
response.json()

# Rename File
headers = get_headers()
response = post(f'{BASE_URL}/files/update', headers = headers,
                data = {
                    "projectId": project_id,
                    "name": "utils.py",
                    "newName": "utils2.py"
                })
response.json()

# Delete File
headers = get_headers()
response = post(f'{BASE_URL}/files/delete', headers = headers, data = { "projectId": project_id, "name": "utils2.py" })
response.json()
