The following example demonstates uploading, getting, deleting, and listing Object Store objects through the cloud API.

from base64 import b64encode
from hashlib import sha256
from time import time
from requests import get, post

USER_ID = 0
ORG_ID = ""
API_TOKEN = '...'
BASE_URL = 'https://www.quantconnect.com/api/v2/'
key = "api_test"
org_id = "..."

def get_headers():
    # Get timestamp
    timestamp = f'{int(time())}'
    time_stamped_token = f'{API_TOKEN}:{timestamp}'.encode('utf-8')

    # Get hased API token
    hashed_token = sha256(time_stamped_token).hexdigest()
    authentication = f'{USER_ID}:{hashed_token}'.encode('utf-8')
    authentication = b64encode(authentication).decode('ascii')

    # Create headers dictionary.
    return {
        'Authorization': f'Basic {authentication}',
        'Timestamp': timestamp
    }

# Upload Object Store File
headers = get_headers()
response = post(f'{BASE_URL}/object/set', headers = headers, 
                data = { "organizationId": org_id, "key": key },
                files = { "objectData": b"Hello, world!" })
response.json()

# Get Object Store Metadata
headers = get_headers()
response = post(f'{BASE_URL}/object/properties', headers = headers, data = { "organizationId": org_id, "key": key })
response.json()

# Get Object Store File
headers = get_headers()
response = post(f'{BASE_URL}/object/get', headers = headers, data = { "organizationId": org_id, "keys": [key] })
response.json()

# Delete Object Store File
headers = get_headers()
response = post(f'{BASE_URL}/object/delete', headers = headers, data = { "organizationId": org_id, "key": key })
response.json()

# List Object Store Files
path = ""
headers = get_headers()
response = post(f'{BASE_URL}/object/delete', headers = headers, data = { "organizationId": org_id, "path": path })
response.json()
