The following example demonstates reading backtest statistics, chart, orders, insights, and report through the cloud API.

from base64 import b64encode
from hashlib import sha256
from time import time
from requests import get, post

USER_ID = 0
ORG_ID = ""
API_TOKEN = '...'
BASE_URL = 'https://www.quantconnect.com/api/v2/'
project_id = 12345678
backtest_id = "..."

def get_headers():
    # Get timestamp
    timestamp = f'{int(time())}'
    time_stamped_token = f'{API_TOKEN}:{timestamp}'.encode('utf-8')

    # Get hased API token
    hashed_token = sha256(time_stamped_token).hexdigest()
    authentication = f'{USER_ID}:{hashed_token}'.encode('utf-8')
    authentication = b64encode(authentication).decode('ascii')

    # Create headers dictionary.
    return {
        'Authorization': f'Basic {authentication}',
        'Timestamp': timestamp
    }

# Read Backtest Statistics
headers = get_headers()
response = post(f'{BASE_URL}/backtests/read', headers = headers, data = { "projectId": project_id, "backtestId": backtest_id })
response.json()

# Read Backtest Charts
chart_name = "Strategy Equity"
headers = get_headers()
response = post(f'{BASE_URL}/backtests/charts/read', headers = headers
                data = {
                    "projectId": project_id,
                    "backtestId": backtest_id,
                    "name": chart_name,
                    "count": 100
                    "start": 0,
                    "end": 100
                })
response.json()

# Read Backtest Orders
headers = get_headers()
response = post(f'{BASE_URL}/backtests/orders/read', headers = headers,
                data = {
                    "projectId": project_id,
                    "backtestId": backtest_id,
                    "start": 0,
                    "end": 100
                })
response.json()

# Read Backtest Insights
headers = get_headers()
response = post(f'{BASE_URL}/backtests/read/insights', headers = headers,
                data = {
                    "projectId": project_id,
                    "backtestId": backtest_id,
                    "start": 0,
                    "end": 100
                })
response.json()

# Read Backtest Report
headers = get_headers()
response = post(f'{BASE_URL}/backtests/read/report', headers = headers,
                data = {
                    "projectId": project_id,
                    "backtestId": backtest_id
                })
response.json()
