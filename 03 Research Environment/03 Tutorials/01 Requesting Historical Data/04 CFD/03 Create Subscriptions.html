Follow these steps to subscribe to a CFD security:

  1. Load the required assembly files and data types.
    2. #load "../Initialize.csx"
#load "../QuantConnect.csx"

using QuantConnect;
using QuantConnect.Data;
using QuantConnect.Algorithm;
using QuantConnect.Research;
  2. Instantiate a QuantBook.
    3. var qb = new QuantBook();
    qb = QuantBook()
  3. Call the AddCfd method with a ticker and resolution.
    4. var cfd = qb.AddCfd("SPX500USD", Resolution.Minute);
    cfd = qb.AddCfd("SPX500USD", Resolution.Minute)

    If you do not pass a resolution argument, Resolution.Minute is used by default.

  4. Save a reference to the CFD Symbol.
    5. var symbol = cfd.Symbol;
    symbol = cfd.Symbol
