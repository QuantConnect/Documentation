Follow these steps to subscribe to a Forex security:

  1. Load the required assembly files and data types.
    2. #load "../Initialize.csx"
#load "../QuantConnect.csx"

using QuantConnect;
using QuantConnect.Data;
using QuantConnect.Algorithm;
using QuantConnect.Research;
  2. Instantiate a QuantBook.
    3. var qb = new QuantBook();
    qb = QuantBook()
  3. Call the AddForex method with a ticker and resolution.
    4. var forex = qb.AddForex("EURUSD", Resolution.Daily);
    forex = qb.AddForex("EURUSD", Resolution.Daily)

    If you do not pass a resolution argument, Resolution.Minute is used by default.

  4. Save a reference to the Forex Symbol.
    5. var symbol = forex.Symbol;
    symbol = forex.Symbol
