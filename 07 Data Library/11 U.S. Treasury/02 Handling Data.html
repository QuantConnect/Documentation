U.S. Treasury yield curve data is reported in percentages with the value being between 0.0 - 100.0. To view the end-of-day interest rate statistics, visit the Daily Treasury Yield Curve Rates page

All fields with the exception of BaseData fields contained within $[USTreasuryYieldCurveRate,T:QuantConnect.Data.Custom.USTreasury] can potentially be null values. Be sure to check for null values before you continue using the data.

In this example, we check for null values before using the data, as well as calculating if there's a yield curve inversion: 

public override void OnData(Slice data)
{
    if (!data.ContainsKey(_symbol))
    {
        return;
    }

    // Get the data using slice.Get<T> instead of data[_symbol] so that null values don't get set to default(decimal)
    var rates = slice.Get<USTreasuryYieldCurveRate>().First().Value;

    // Check for null values first
    if (!rates.TwoYear.HasValue || !rates.TenYear.HasValue)
    {
        return;
    }

    // Now that we're certain that we have the values, see if we have a yield curve inversion
    if (rates.TwoYear > rates.TenYear) 
    {
        Debug($"{Time} - Yield curve inversion!");
    }
}
        
def OnData(data):
    if not data.ContainsKey(self.symbol):
        return
    
    rates = data[self.symbol]

    # Check for None values first
    if rates.TwoYear is None or rates.TenYear is None:
        return
    
    # Now that we're certain that we have the values, see if we have a yield curve inversion
    if rates.TwoYear > rates.TenYear:
        self.Debug(f"{str(self.Time)} - Yield curve inversion!")
