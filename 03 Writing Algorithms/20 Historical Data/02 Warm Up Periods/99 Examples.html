Example 1: Warm Up Indicator

You can warm up an indicator by the number of sample it needs.

private RelativeStrengthIndex _rsi;
    
public override void Initialize()
{
    var spy = AddEquity("SPY", Resolution.Minute).Symbol;
    _rsi = RSI(spy, 14, Resolution.Daily);
    SetWarmUp(14 * 390);
    // alterantive: WarmUpIndicator(_rsi);
}
def initialize(self):
    spy = self.add_equity("SPY", Resolution.MINUTE).symbol
    self._rsi = self.rsi(spy, 14, Resolution.DAILY)
    self,set_warm_up(14 * 390)
    # alterantive: self.warm_up_indicator(self._rsi)

Example 2: Train Machine Learning Model On Warm Up Finishes

Train your ML model when the warm up process is completed and your indicators/RollingWindow are ready.

private RelativeStrengthIndex _rsi;
    
public override void Initialize()
{
    var spy = AddEquity("SPY", Resolution.Minute).Symbol;
    _rsi = RSI(spy, 14, Resolution.Daily);
    _rsi.Window.Size = 20
    SetWarmUp(34 * 390);
}

public override void OnWarmupFinished()
{
    Train(MyTrainingMethod);
}
def initialize(self):
    spy = self.add_equity("SPY", Resolution.MINUTE).symbol
    self._rsi = self.rsi(spy, 14, Resolution.DAILY)
    self._rsi.window.size = 20
    self,set_warm_up(34 * 390)
  
def on_warmup_finished(self):
    self.train(self.my_training_method)
