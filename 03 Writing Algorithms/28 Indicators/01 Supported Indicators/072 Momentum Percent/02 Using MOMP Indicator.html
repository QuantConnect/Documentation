To create an automatic indicator for MomentumPercent, call the MOMPmomp helper method from the QCAlgorithm class. The MOMPmomp method creates a MomentumPercent object, hooks it up for automatic updates, and returns it so you can used it in your algorithm. In most cases, you should call the helper method in the Initializeinitialize method.

public class MomentumPercentAlgorithm : QCAlgorithm
{
    private Symbol _symbol;
    private MomentumPercent _momp;

    public override void Initialize()
    {
        _symbol = AddEquity("SPY", Resolution.Daily).Symbol;
        _momp = MOMP(_symbol, 20);
    }

    public override void OnData(Slice data)
    {

        if (_momp.IsReady)
        {
            // The current value of _momp is represented by itself (_momp)
            // or _momp.Current.Value
            Plot("MomentumPercent", "momp", _momp);
        }
    }
}
class MomentumPercentAlgorithm(QCAlgorithm):
    def initialize(self) -> None:
        self._symbol = self.add_equity("SPY", Resolution.DAILY).symbol
        self._momp = self.momp(self._symbol, 20)

    def on_data(self, slice: Slice) -> None:

        if self._momp.is_ready:
            # The current value of self._momp is represented by self._momp.current.value
            self.plot("MomentumPercent", "momp", self._momp.current.value)

You can manually create a MomentumPercent indicator, so it doesn't automatically update. Manual indicators let you update their values with any data you choose.

Updating your indicator manually enables you to control when the indicator is updated and what data you use to update it. To manually update the indicator, call the Updateupdate method. The indicator will only be ready after you prime it with enough data.

public class MomentumPercentAlgorithm : QCAlgorithm
{
    private Symbol _symbol;
    private MomentumPercent _momentumpercent;

    public override void Initialize()
    {
        _symbol = AddEquity("SPY", Resolution.Daily).Symbol;
        _momentumpercent = new MomentumPercent(20);
    }

    public override void OnData(Slice data)
    {
        if (data.Bars.TryGetValue(_symbol, out var bar))
            _momentumpercent.Update(bar.EndTime, bar.Close);

        if (_momentumpercent.IsReady)
        {
            // The current value of _momentumpercent is represented by itself (_momentumpercent)
            // or _momentumpercent.Current.Value
            Plot("MomentumPercent", "momentumpercent", _momentumpercent);
        }
    }
}
class MomentumPercentAlgorithm(QCAlgorithm):
    def initialize(self) -> None:
        self._symbol = self.add_equity("SPY", Resolution.DAILY).symbol
        self._momentumpercent = MomentumPercent(20)

    def on_data(self, slice: Slice) -> None:
        bar = slice.bars.get(self._symbol)
        if bar:
            self._momentumpercent.update(bar.end_time, bar.close)

        if self._momentumpercent.is_ready:
            # The current value of self._momentumpercent is represented by self._momentumpercent.current.value
            self.plot("MomentumPercent", "momentumpercent", self._momentumpercent.current.value)
