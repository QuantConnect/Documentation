To create an automatic indicator for Covariance, call the COVcov helper method from the QCAlgorithm class. The COVcov method creates a Covariance object, hooks it up for automatic updates, and returns it so you can used it in your algorithm. In most cases, you should call the helper method in the Initializeinitialize method.

public class CovarianceAlgorithm : QCAlgorithm
{
    private Symbol _symbol,_reference;
    private Covariance _cov;

    public override void Initialize()
    {
        _symbol = AddEquity("QQQ", Resolution.Daily).Symbol;
        _reference = AddEquity("SPY", Resolution.Daily).Symbol;
        _cov = COV([_symbol, _reference]);
    }

    public override void OnData(Slice data)
    {

        if (_cov.IsReady)
        {
            // The current value of _cov is represented by itself (_cov)
            // or _cov.Current.Value
            Plot("Covariance", "cov", _cov);
        }
    }
}
class CovarianceAlgorithm(QCAlgorithm):
    def initialize(self) -> None:
        self._symbol = self.add_equity("QQQ", Resolution.DAILY).symbol
        self._reference = self.add_equity("SPY", Resolution.DAILY).symbol
        self._cov = self.cov([self._symbol, self._reference])

    def on_data(self, slice: Slice) -> None:

        if self._cov.is_ready:
            # The current value of self._cov is represented by self._cov.current.value
            self.plot("Covariance", "cov", self._cov.current.value)

You can manually create a Covariance indicator, so it doesn't automatically update. Manual indicators let you update their values with any data you choose.

Updating your indicator manually enables you to control when the indicator is updated and what data you use to update it. To manually update the indicator, call the Updateupdate method. The indicator will only be ready after you prime it with enough data.

public class CovarianceAlgorithm : QCAlgorithm
{
    private Symbol _symbol,_reference;
    private Covariance _covariance;

    public override void Initialize()
    {
        _symbol = AddEquity("QQQ", Resolution.Daily).Symbol;
        _reference = AddEquity("SPY", Resolution.Daily).Symbol;
        _covariance = new Covariance(_symbol, _reference, 5);
    }

    public override void OnData(Slice data)
    {
        if (data.Bars.TryGetValue(_symbol, out var bar))
            _covariance.Update(bar);
        if (data.Bars.TryGetValue(_reference, out bar))
            _covariance.Update(bar);

        if (_covariance.IsReady)
        {
            // The current value of _covariance is represented by itself (_covariance)
            // or _covariance.Current.Value
            Plot("Covariance", "covariance", _covariance);
        }
    }
}
class CovarianceAlgorithm(QCAlgorithm):
    def initialize(self) -> None:
        self._symbol = self.add_equity("QQQ", Resolution.DAILY).symbol
        self._reference = self.add_equity("SPY", Resolution.DAILY).symbol
        self._covariance = Covariance(self._symbol, self._reference, 5)

    def on_data(self, slice: Slice) -> None:
        bar = slice.bars.get(self._symbol)
        if bar:
            self._covariance.update(bar)
        bar = slice.bars.get(self._reference)
        if bar:
            self._covariance.update(bar)

        if self._covariance.is_ready:
            # The current value of self._covariance is represented by self._covariance.current.value
            self.plot("Covariance", "covariance", self._covariance.current.value)
