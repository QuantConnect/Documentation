To create an automatic indicators for InternalBarStrength, call the IBSibs helper method from the QCAlgorithm class. The IBSibs method creates a InternalBarStrength object, hooks it up for automatic updates, and returns it so you can used it in your algorithm. In most cases, you should call the helper method in the Initializeinitialize method.

public class InternalBarStrengthAlgorithm : QCAlgorithm
{
    private Symbol _symbol;
    private InternalBarStrength _ibs;

    public override void Initialize()
    {
        _symbol = AddEquity("SPY", Resolution.Daily).Symbol;
        _ibs = IBS(_symbol);
    }

    public override void OnData(Slice data)
    {

        if (_ibs.IsReady)
        {
            // The current value of _ibs is represented by itself (_ibs)
            // or _ibs.Current.Value
            Plot("InternalBarStrength", "ibs", _ibs);
        }
    }
}

You can manually create a InternalBarStrength indicator, so it doesn't automatically update. Manual indicators let you update their values with any data you choose.

Updating your indicator manually enables you to control when the indicator is updated and what data you use to update it. To manually update the indicator, call the Updateupdate method. The indicator will only be ready after you prime it with enough data.

public class InternalBarStrengthAlgorithm : QCAlgorithm
{
    private Symbol _symbol;
    private InternalBarStrength _internalbarstrength;

    public override void Initialize()
    {
        _symbol = AddEquity("SPY", Resolution.Daily).Symbol;
        _internalbarstrength = new InternalBarStrength();
    }

    public override void OnData(Slice data)
    {
        if (data.Bars.TryGetValue(_symbol, out var bar))
            _internalbarstrength.Update(bar);

        if (_internalbarstrength.IsReady)
        {
            // The current value of _internalbarstrength is represented by itself (_internalbarstrength)
            // or _internalbarstrength.Current.Value
            Plot("InternalBarStrength", "internalbarstrength", _internalbarstrength);
        }
    }
}
class InternalBarStrengthAlgorithm(QCAlgorithm):
    def initialize(self) -> None:
        self._symbol = self.add_equity("SPY", Resolution.DAILY).symbol
        self._internalbarstrength = InternalBarStrength()

    def on_data(self, slice: Slice) -> None:
        bar = slice.bars.get(self._symbol)
        if bar:
            self._internalbarstrength.update(bar)

        if self._internalbarstrength.is_ready:
            # The current value of self._internalbarstrength is represented by self._internalbarstrength.current.value
            self.plot("InternalBarStrength", "internalbarstrength", self._internalbarstrength.current.value)
