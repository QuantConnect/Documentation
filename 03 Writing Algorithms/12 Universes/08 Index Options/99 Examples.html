Example 1: 30-day expiring Straddle

You can filter option universe to obtain an ATM call and an ATM put contract with minimum 30 days from expiry to form a straddle for volatility trading.

public override void Initialize()
{
    var index = AddIndex("SPX");
    var indexOption = AddIndexOption(index.Symbol, "SPXW");
    indexOption.SetFilter(optionFilterUniverse => optionFilterUniverse.IncludeWeeklys().Straddle(30));
}
def initialize(self):
    index = self.add_index("SPX")
    index_option = self.add_index_option(index.symbol, "SPXW")
    index_option.set_filter(lambda option_filter_universe: option_filter_universe.include_weeklys().straddle(30))
