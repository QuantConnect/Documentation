Example 1: 30-day expiring Straddle

You can filter option universe to obtain an ATM call and an ATM put contract with minimum 30 days from expiry to form a straddle for volatility trading.

public override void Initialize()
{
    var future = AddFuture("ES");
    var futureOption = AddFutureOption(future.Symbol);
    futureOption.SetFilter(optionFilterUniverse => optionFilterUniverse.IncludeWeeklys().Straddle(30));
}
def initialize(self):
    future = self.add_future("ES")
    future_option = self.add_future_option(future.symbol)
    future_option.set_filter(lambda option_filter_universe: option_filter_universe.include_weeklys().straddle(30))
