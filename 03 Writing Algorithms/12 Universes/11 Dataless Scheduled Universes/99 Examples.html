Example 1: Quarter End Selection

In this example, we select SPY if the current month is a quarter's end to ride on the possible quarter ending accounting period. Otherwise, we return an empty universe.

AddUniverse(new ScheduledUniverse(DateRules.MonthStart(), TimeRules.At(8, 0), (dt) => 
{
    if (dt.Month % 3 == 0)
    {
        return new[] { Symbol.Create("SPY", SecurityType.Equity, Market.USA) };
    }
    return Enumrable.Empty<Symbol>();
}));
self.add_universe(
    ScheduledUniverse(
        self.date_rules.month_start(), 
        self.time_rules.at(8, 0), 
        self._select_symbols
    )
)
	
def _select_symbols(self, dt: datetime) -> List[Symbols]:
    if dt.month % 3 == 0:
        return [Symbol.create("SPY", SecurityType.EQUITY, Market.USA)]
    return []

Example 2: Third Week VIX

The standard options are expiring at the third week's end of each month. You can filter the universe to select VIX related products on the third week to trade the foreseeable increasing volatility.

_month = -1;
_week = -1;
    
AddUniverse(new ScheduledUniverse(DateRules.WeekStart(), TimeRules.At(8, 0), (dt) => 
{
    if (dt.Month == _month)
    {
        if (++_week == 3)
        {
          return new[] { Symbol.Create("VXZ", SecurityType.Equity, Market.USA) };
        }
        return Enumrable.Empty<Symbol>();
    }
    
    _month = dt.Month;
    _week = 0;
    return Enumrable.Empty<Symbol>();
}));
self.month = -1
self.week = -1

self.add_universe(
    ScheduledUniverse(
        self.date_rules.week_start(), 
        self.time_rules.at(8, 0), 
        self._select_symbols
    )
)
	
def _select_symbols(self, dt: datetime) -> List[Symbols]:
    if dt.month == self.month:
        self.week += 1
        if self.week == 3:
            return [Symbol.create("VXZ", SecurityType.EQUITY, Market.USA)]
        return []
    
    self.month = dt.month
    self.week = 0
    return []
