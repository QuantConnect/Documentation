To select a dynamic universe of US Equities based on Wikipedia Page Views data, call the AddUniverse method with the QuiverWikipediaUniverse class and a selection function.

def initialize(self) -> None:
    self.universe = self.add_universe(QuiverWikipediaUniverse, self.universe_selection)

def universe_selection(self, alt_coarse: List[QuiverWikipediaUniverse]) -> List[Symbol]:
    return [d.symbol for d in alt_coarse \
                if d.page_views > 100 \
                and d.week_percent_change < 0.2]
private Universe _universe;
public override void Initialize()
{
    _universe = AddUniverse<QuiverWikipediaUniverse>(altCoarse =>
    {
        return from d in altCoarse
                where d.PageViews > 100m && d.MonthPercentChange > 0.2m
                select d.Symbol;
    });
}
