The Brain Sentiment Indicator dataset enables you to incorporate sentiment from financial news sources into your strategies. Examples include the following strategies:

  • Buying when the public sentiment for a security is increasing
  • Short selling when the public sentiment for a security is decreasing
  • Scaling the position sizing of securities based on how many times they are mentioned in financial news articles
  • Sector rotation based on news sentiment

Classic Algorithm Example

The following example algorithm buys Apple when the 30-day Brain Sentiment indicator increases. Otherwise, it remains in cash.

from AlgorithmImports import *


class BrainSentimentDataAlgorithm(QCAlgorithm):
    latest_sentiment_value = None
    target_holdings = 0

    def initialize(self) -> None:
        self.set_start_date(2024, 9, 1)
        self.set_end_date(2024, 12, 31)
        self.set_cash(100000)
        # Request the processed longer term sentiment score data for sentiment trading.
        self.aapl = self.add_equity("AAPL", Resolution.DAILY)
        self.dataset_symbol = self.add_data(BrainSentimentIndicator30Day, self.aapl).symbol
        history = self.history(self.dataset_symbol, 252, Resolution.DAILY)
        self.debug(f"We got {len(history)} items from our history request for {self.dataset_symbol}")
        if history.empty:
            return
        # Warm up historical sentiment values so the signal is immediately tradable.
        previous_sentiment_values = history.loc[self.dataset_symbol].sentiment.values
        for sentiment in previous_sentiment_values:
            self.update(sentiment)

    def update(self, sentiment: float) -> None:
        # Compare the last sentiment score to buy when sentiment increases.
        if self.latest_sentiment_value is not None:
            self.target_holdings = int(sentiment > self.latest_sentiment_value)
        self.latest_sentiment_value = sentiment

    def on_data(self, data: Slice) -> None:
        # Update trade direction from the latest sentiment data.
        if self.dataset_symbol in data:
            sentiment = data[self.dataset_symbol].sentiment
            self.update(sentiment)
        # Require current equity data to avoid stale fills.
        if self.aapl not in data or data[self.aapl] is None:
            return
        # Buy when sentiment increases, otherwise liquidate the equity.
        if self.target_holdings != self.portfolio.invested:
            self.set_holdings(self.aapl, self.target_holdings)
public class BrainSentimentDataAlgorithm : QCAlgorithm
{
    private Equity _aapl;
    private Symbol _datasetSymbol;
    private decimal? _latestSentimentValue = null;
    private int _targetHoldings = 0;

    public override void Initialize()
    {
        SetStartDate(2024, 9, 1);
        SetEndDate(2024, 12, 31);
        SetCash(100000);
        // Request the processed longer term sentiment score data for sentiment trading.
        _aapl = AddEquity("AAPL", Resolution.Daily);
        _datasetSymbol = AddData<BrainSentimentIndicator30Day>(_aapl.Symbol).Symbol;
        var history = History<BrainSentimentIndicator30Day>(_datasetSymbol, 252, Resolution.Daily);
        Debug($"We got {history.Count()} items from our history request for {_datasetSymbol}");
        // Warm up historical sentiment values so the signal is immediately tradable.
        var previousSentimentValues = history.Select(x => x.Sentiment);
        foreach (var sentiment in previousSentimentValues)
        {
            Update(sentiment);
        }
    }

    public void Update(decimal sentiment)
    {
        // Compare the last sentiment score to buy when sentiment increases.
        if (_latestSentimentValue != null)
        {
            _targetHoldings = sentiment > _latestSentimentValue ? 1 : 0;
        }
        _latestSentimentValue = sentiment;
    }

    public override void OnData(Slice data)
    {
        // Update trade direction from the latest sentiment data.
        if (data.ContainsKey(_datasetSymbol))
        {
            var sentiment = data[_datasetSymbol].Sentiment;
            Update(sentiment);
        }
        // Require current equity data to avoid stale fills.
        if (!data.Bars.ContainsKey(_aapl.Symbol))
        {
            return;
        }
        // Buy when sentiment increases, otherwise liquidate the equity.
        if (_targetHoldings == 1 != Portfolio.Invested)
        {
            SetHoldings(_aapl.Symbol, _targetHoldings);
        }
    }
}

Framework Algorithm Example

The following example algorithm creates a dynamic universe of US Equities that have been mentioned in an article over the last seven days. It then buys the subset of Equities that have increasing sentiment and forms an equal-weighted portfolio.

from AlgorithmImports import *


class BrainSentimentDataAlgorithm(QCAlgorithm):
    def initialize(self) -> None:
        self.set_start_date(2024, 9, 1)
        self.set_end_date(2024, 12, 31)
        self.set_cash(100000)
        self.settings.seed_initial_prices = True
        self.universe_settings.resolution = Resolution.HOUR
        # Filter base on sentiment data
        self.add_universe(BrainSentimentIndicatorUniverse, self.universe_selection)        
        self.add_alpha(BrainSentimentAlphaModel())        
        self.set_portfolio_construction(EqualWeightingPortfolioConstructionModel())         
        self.add_risk_management(NullRiskManagementModel())        
        self.set_execution(ImmediateExecutionModel())

    def universe_selection(self, alt_coarse: List[BrainSentimentIndicatorUniverse]) -> List[Symbol]:
        # Select the stocks with the most sentiment news in the last 7 days
        filtered = [d for d in alt_coarse if d.sentimental_article_mentions_7_days]
        filtered = sorted(filtered, key=lambda d: d.sentimental_article_mentions_7_days)[-50:]
        return [d.symbol for d in filtered]


class BrainSentimentAlphaModel(AlphaModel):
    _securities = []

    def update(self, algorithm: QCAlgorithm, slice: Slice) -> List[Insight]:
        insights = []
        for dataset_symbol, data_point in slice.get(BrainSentimentIndicator30Day).items():
            security = algorithm.securities[dataset_symbol.underlying]
            target_direction = InsightDirection.FLAT
            if security.latest_sentiment_value and data_point.sentiment > security.latest_sentiment_value:
                target_direction = InsightDirection.UP
            security.latest_sentiment_value = data_point.sentiment
            # Buy if sentiment increase, liquidate otherwise to ride on the popularity of the equity
            if security.price and target_direction == InsightDirection.UP and not security.invested:
                insights.append(Insight.price(security, timedelta(252), target_direction))
        return insights
                
    def on_securities_changed(self, algorithm: QCAlgorithm, changes: SecurityChanges) -> None:
        for security in changes.added_securities:
            # Subscribe to custom data and store dataset symbol
            dataset = algorithm.add_data(BrainSentimentIndicator30Day, security, Resolution.DAILY)
            security.brain_sentiment_data = dataset.symbol
            security.latest_sentiment_value = None
            # Historical data
            history = algorithm.history(dataset, 252, Resolution.DAILY)
            if not history.empty:
                # Warm up historical sentiment values, cache for comparing last sentiment score to trade
                previous_sentiment_values = history.loc[security.brain_sentiment_data].sentiment.values
                for sentiment in previous_sentiment_values:
                    security.latest_sentiment_value = sentiment
            self._securities.append(security)
        for security in changes.removed_securities:
            # Unsubscribe from the Brain Sentiment feed
            algorithm.remove_security(security.brain_sentiment_data)
            # Remove from tracking
            if security in self._securities:
                self._securities.remove(security)
public class BrainSentimentDataAlgorithm : QCAlgorithm
{
    public override void Initialize()
    {
        SetStartDate(2024, 9, 1);
        SetEndDate(2024, 12, 31);
        SetCash(100000);
        Settings.SeedInitialPrices = true;
        UniverseSettings.Resolution = Resolution.Hour;
        // Filter base on sentiment data
        AddUniverse<BrainSentimentIndicatorUniverse>(
            altCoarse => altCoarse
            .Select(d => d as BrainSentimentIndicatorUniverse)
            .Where(d => d.SentimentalArticleMentions7Days != 0m)
            .OrderBy(d => d.SentimentalArticleMentions7Days)
            .TakeLast(50)
            .Select(d => d.Symbol)
        );
        AddAlpha(new BrainSentimentAlphaModel());
        SetPortfolioConstruction(new EqualWeightingPortfolioConstructionModel());
        AddRiskManagement(new NullRiskManagementModel());
        SetExecution(new ImmediateExecutionModel());
    }
}


public class BrainSentimentAlphaModel : AlphaModel
{
    private readonly List<Security> _securities = [];

    public override IEnumerable<Insight> Update(QCAlgorithm algorithm, Slice slice)
    {
        var insights = new List<Insight>();
        foreach (var (datasetSymbol, dataPoint) in slice.Get<BrainSentimentIndicator30Day>())
        {
            var security = algorithm.Securities[datasetSymbol.Underlying];
            var targetDirection = InsightDirection.Flat;
            var latestSentimentValue = security.Get<decimal?>("LatestSentimentValue");
            if (latestSentimentValue.HasValue && dataPoint.Sentiment > latestSentimentValue.Value)
            {
                targetDirection = InsightDirection.Up;
            }
            security.Set("LatestSentimentValue", dataPoint.Sentiment);
            // Buy if sentiment increase, liquidate otherwise to ride on the popularity of the equity
            if (security.Price != 0 && targetDirection == InsightDirection.Up && !security.Holdings.Invested)
            {
                insights.Add(Insight.Price(security.Symbol, TimeSpan.FromDays(252), targetDirection));
            }
        }
        return insights;
    }

    public override void OnSecuritiesChanged(QCAlgorithm algorithm, SecurityChanges changes)
    {
        foreach (var security in changes.AddedSecurities)
        {
            // Subscribe to custom data and store dataset symbol
            var dataset = algorithm.AddData<BrainSentimentIndicator30Day>(security.Symbol, Resolution.Daily);
            security.Set("BrainSentimentData", dataset.Symbol);
            security.Set("LatestSentimentValue", null);
            // Historical data
            var history = algorithm.History<BrainSentimentIndicator30Day>(dataset.Symbol, 252, Resolution.Daily).ToList();
            if (history.Count() != 0)
            {
                // Warm up historical sentiment values, cache for comparing last sentiment score to trade
                foreach (var sentiment in history.Select(dataPoint => dataPoint.Sentiment))
                {
                    security.Set("LatestSentimentValue", sentiment);
                }
            }
            _securities.Add(security);
        }

        foreach (var security in changes.RemovedSecurities)
        {
            // Unsubscribe from the Brain Sentiment feed
            algorithm.RemoveSecurity(security.Get<Symbol>("BrainSentimentData"));
            // Remove from tracking
            if (_securities.Contains(security))
            {
                _securities.Remove(security);
            }
        }
    }
}
