To add US Cash Indices data to your algorithm, call the AddIndex method. Save a reference to the Index Symbol so you can access the data later in your algorithm.

class USCashIndexAlgorithm(QCAlgorithm):

    def initialize(self) -> None:
        self.set_start_date(2020, 6, 1)
        self.set_end_date(2021, 6, 1)
        self.set_cash(100000)

        self.symbol = self.add_index("VIX").symbol

namespace QuantConnect
{
    public class USCashIndexAlgorithm : QCAlgorithm
    {
        private Symbol _symbol;
        
        public override void Initialize()
        {
            SetStartDate(2020, 6, 1);
            SetEndDate(2021, 6, 1);
            SetCash(100000);
            
            _symbol = AddIndex("VIX").Symbol;
        }
    }
}
