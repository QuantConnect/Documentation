Example 1: Only Within 1 Hour Of Market Open And Close

Some traders prefer to use only data of 1 hour after market open and before market close, since they have the most volume.

security.SetDataFilter(new CustomDataFilter())

private class CustomDataFilter : ISecurityDataFilter
{
    public bool Filter(Security vehicle, BaseData data)
    {
        if (data.Time.Hour >= 10 && data.Time.Hour <= 14)
        {
            return false;
        }
        return true;
    }
}
security.set_data_filter(CustomDataFilter())

class CustomDataFilter(SecurityDataFilter):
    def filter(self, vehicle: Security, data: BaseData) -> bool:
        if 14 >= data.time.hour >= 10:
            return False
        return True

Example 2: Use Only Quote Ticks

You can use only quote ticks but not trade ticks for execution model training or more.

security.SetDataFilter(new CustomDataFilter())

private class CustomDataFilter : ISecurityDataFilter
{
    public bool Filter(Security vehicle, BaseData data)
    {
        var tick = data as Tick;
        
        if (tick.TickType == TickType.Trade)
        {
            return false;
        }
        return true;
    }
}
security.set_data_filter(CustomDataFilter())

class CustomDataFilter(SecurityDataFilter):
    def filter(self, vehicle: Security, data: BaseData) -> bool:
        tick = Tick(data)
        
        if tick.tick_type == TickType.TRADE:
            return False
        return True

Example 3: Train Model For Suspicious Data

You can also train a model to detect suspicious data feed and filter them

security.SetDataFilter(new CustomDataFilter())

private class CustomDataFilter : ISecurityDataFilter
{
    public bool Filter(Security vehicle, BaseData data)
    {
        var tick = data as Tick;
        
        if (_model.IsSuspicious(tick))
        {
            return false;
        }
        return true;
    }
}
security.set_data_filter(CustomDataFilter())

class CustomDataFilter(SecurityDataFilter):
    def filter(self, vehicle: Security, data: BaseData) -> bool:
        tick = Tick(data)
        
        if self.model.is_suspicious(tick):
            return False
        return True
