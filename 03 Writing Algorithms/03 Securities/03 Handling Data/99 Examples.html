Example 1: Use QuoteBar Data

You can use mid-price of quote bar data to update indicator of illiquid assets for accuracy.

public override void OnData(Slice slice)
{
    if (slice.QuoteBars.ContainsKey(_symbol))
    {
        var quoteBar = slice.QuoteBars[_symbol];
        var midPrice = (quoteBar.Bid.Price + quoteBar.Ask.Price) * 0.5m;
        _indicator.Update(slice.Time, midPrice);
    }
}
def on_data(self, slice: Slice) -> None:
    if slice.quote_bars.contains_key(self._symbol):
        quote_bar = slice.quote_bars[self._symbol]
        mid_price = (quote_bar.bid.price + quote_bar.ask.price) * 0.5
        _indicator.update(slice.time, mid_price)

Example 2: Use Tick Data

You can use tick data to obtain the instant bid and ask size of the latest tick.

public override void OnData(Slice slice)
{
    if (slice.Ticks.ContainsKey(_symbol))
    {
        var tick = slice.Ticks[_symbol];
        if (tick.TickType == TickType.Quote)
        {
            var bidSize = tick.BidSize;
            var askSize = tick.AskSize;
        }
    }
}
def on_data(self, slice: Slice) -> None:
    if slice.ticks.contains_key(self._symbol):
        tick = slice.ticks[self._symbol]
        if tick.tick_type == TickType.QUOTE:
            bid_size = tick.bid_size
            ask_size = tick.ask_size
